  namespace units {
    template<typename Y> class absolute;

    // add a relative value to an absolute one 
    template<typename Y> 
      BOOST_CONSTEXPR absolute< Y > 
      operator+(const absolute< Y > & aval, const Y & rval);

    // add a relative value to an absolute one 
    template<typename Y> 
      BOOST_CONSTEXPR absolute< Y > 
      operator+(const Y & rval, const absolute< Y > & aval);

    // subtract a relative value from an absolute one

      BOOST_CONSTEXPR absolute< Y >

      BOOST_CONSTEXPR Y
